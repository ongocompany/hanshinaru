# AI 저작툴 (AI Writer) 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 한시나루 콘텐츠를 AI 보조로 작성하고 Supabase DB에 저장하는 단일 페이지 저작 도구 구현

**Architecture:** tools/ai-writer/ 디렉토리에 index.html + 모듈별 JS 파일로 구성. OpenAI 호환 chat completions API로 AI 호출하고, Supabase REST API로 DB CRUD. 모든 설정은 localStorage에 저장.

**Tech Stack:** Vanilla HTML/CSS/JS, OpenAI 호환 API (SSE 스트리밍), Supabase REST API (fetch 직접 호출)

**Spec:** `docs/spec/2026-04-13-ai-writer-design.md`

---

## 파일 구조

```
tools/ai-writer/
├── index.html          # 메인 HTML (레이아웃, 스타일)
├── js/
│   ├── app.js          # 앱 초기화, 이벤트 바인딩, 모드 전환
│   ├── api.js          # 멀티 프로바이더 AI API 클라이언트 (SSE 스트리밍)
│   ├── editor.js       # 에디터 코어 (섹션 관리, contenteditable)
│   ├── toolbar.js      # 플로팅 툴바 + 말풍선, 슬래시 커맨드
│   ├── prompt.js       # 프롬프트 조합 (시스템 + 유저)
│   ├── db.js           # Supabase REST CRUD (4개 테이블)
│   └── settings.js     # API 설정 UI, 프리셋/칩 관리, localStorage
```

각 파일의 책임:
- **index.html**: 전체 레이아웃 HTML + CSS. JS 파일을 `<script>` 태그로 로드
- **app.js**: DOM 초기화, 모드 탭 전환, "초안 생성" 버튼 핸들러, 전체 워크플로우 오케스트레이션
- **api.js**: `AIClient` 클래스 — 프로바이더별 base URL/헤더 분기, SSE 스트리밍 fetch, 연결 테스트
- **editor.js**: `Editor` 클래스 — 4개 모드별 섹션 렌더링, 섹션 추가/삭제/순서변경, contenteditable 관리
- **toolbar.js**: `Toolbar` 클래스 — 텍스트 선택 감지 → 플로팅 툴바, `/` 입력 감지 → 명령 팔레트
- **prompt.js**: `PromptBuilder` — 모드/프리셋/구조/분량/지시사항을 조합해 system+user 프롬프트 생성, 섹션 구분자 파싱
- **db.js**: `DB` 클래스 — poems/poets/history_cards/articles 각 테이블의 CRUD, 모드별 저장/불러오기
- **settings.js**: `Settings` — API 설정 모달, 콘텐츠 유형 칩 관리, 문체 프리셋 관리, localStorage 래퍼

```html
  }
```

```javascript
  return { init, toast, switchMode, get currentMode() { return currentMode; } };
})();
```

```javascript
// tools/ai-writer/js/api.js
'use strict';

const AIClient = (() => {

  function getHeaders(config) {
    const headers = {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${config.apiKey}`,
    };
    // OpenRouter는 추가 헤더 권장
    if (config.provider === 'openrouter') {
      headers['HTTP-Referer'] = window.location.href;
      headers['X-Title'] = 'hanshinaru-ai-writer';
    }
    return headers;
  }

  function getBaseUrl(config) {
    if (config.baseUrl) return config.baseUrl;
    const provider = Settings.DEFAULT_PROVIDERS.find(p => p.id === config.provider);
    return provider ? provider.baseUrl : '';
  }

  async function testConnection(config) {
    const baseUrl = getBaseUrl(config);
    const res = await fetch(`${baseUrl}/chat/completions`, {
      method: 'POST',
      headers: getHeaders(config),
      body: JSON.stringify({
        model: config.model,
        messages: [{ role: 'user', content: 'ping' }],
        max_tokens: 5,
      }),
    });
    if (!res.ok) {
      const err = await res.text();
      throw new Error(`${res.status}: ${err}`);
    }
    return true;
  }

  async function streamChat(config, { system, user }, onChunk) {
    const baseUrl = getBaseUrl(config);
    const res = await fetch(`${baseUrl}/chat/completions`, {
      method: 'POST',
      headers: getHeaders(config),
      body: JSON.stringify({
        model: config.model,
        messages: [
          { role: 'system', content: system },
          { role: 'user', content: user },
        ],
        stream: true,
        temperature: 0.7,
        max_tokens: 8192,
      }),
    });

    if (!res.ok) {
      const err = await res.text();
      throw new Error(`API 오류 ${res.status}: ${err}`);
    }

    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split('\n');
      buffer = lines.pop(); // 마지막 불완전한 줄은 버퍼에 유지

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        const data = line.slice(6).trim();
        if (data === '[DONE]') return;

        try {
          const parsed = JSON.parse(data);
          const content = parsed.choices?.[0]?.delta?.content;
          if (content) onChunk(content);
        } catch {
          // 파싱 실패는 무시 (불완전한 JSON)
        }
      }
    }
  }

  // 비스트리밍 호출 (섹션 재생성 등 짧은 요청)
  async function chat(config, { system, user }) {
    const baseUrl = getBaseUrl(config);
    const res = await fetch(`${baseUrl}/chat/completions`, {
      method: 'POST',
      headers: getHeaders(config),
      body: JSON.stringify({
        model: config.model,
        messages: [
          { role: 'system', content: system },
          { role: 'user', content: user },
        ],
        temperature: 0.7,
        max_tokens: 4096,
      }),
    });

    if (!res.ok) {
      const err = await res.text();
      throw new Error(`API 오류 ${res.status}: ${err}`);
    }

    const json = await res.json();
    return json.choices?.[0]?.message?.content || '';
  }

  return { testConnection, streamChat, chat };
})();
```

```javascript
      sections.push({ name, content });

  }
```

```javascript
// tools/ai-writer/js/toolbar.js
'use strict';

const Toolbar = (() => {
  let selectedRange = null;

  const SLASH_COMMANDS = [
    { id: 'continue', label: '이어서 작성', icon: '✨' },
    { id: 'rewrite', label: '다시 쓰기', icon: '📝' },
    { id: 'expand', label: '분량 늘리기', icon: '📏' },
    { id: 'shorten', label: '분량 줄이기', icon: '✂️' },
    { id: 'tone', label: '톤 변경', icon: '🔄' },
  ];

  function init() {
    initFloatingToolbar();
    initSlashCommand();
  }

  // === 플로팅 툴바 ===
  function initFloatingToolbar() {
    const toolbar = document.getElementById('floating-toolbar');
    toolbar.innerHTML = `
      <button data-format="bold">B</button>
      <button data-format="italic">I</button>
      <button data-format="underline">U</button>
      <span style="border-left:1px solid var(--c-text-faint);margin:0 4px;"></span>
      <button class="btn-ai" id="toolbar-ai-btn">AI</button>
    `;

    // 텍스트 선택 감지
    document.getElementById('editor-body').addEventListener('mouseup', handleSelection);
    document.addEventListener('mousedown', e => {
      if (!e.target.closest('.floating-toolbar') && !e.target.closest('.ai-bubble')) {
        hideToolbar();
        hideAiBubble();
      }
    });

    // 서식 버튼
    toolbar.addEventListener('click', e => {
      const format = e.target.dataset.format;
      if (format) {
        document.execCommand(format);
      }
    });

    // AI 버튼 -> 말풍선
    document.getElementById('toolbar-ai-btn').addEventListener('click', showAiBubble);

    // 말풍선 UI
    const bubble = document.getElementById('ai-bubble');
    bubble.innerHTML = `
      <div class="ai-bubble__label">AI에게 지시</div>
      <input class="ai-bubble__input" id="ai-bubble-input" placeholder="예: 좀 더 구체적인 예시를 들어줘">
      <div class="ai-bubble__actions">
        <button id="ai-bubble-apply">적용</button>
      </div>
    `;

    document.getElementById('ai-bubble-apply').addEventListener('click', handleAiBubbleApply);
    document.getElementById('ai-bubble-input').addEventListener('keydown', e => {
      if (e.key === 'Enter') handleAiBubbleApply();
      if (e.key === 'Escape') hideAiBubble();
    });
  }

  function handleSelection() {
    const sel = window.getSelection();
    if (!sel || sel.isCollapsed || !sel.toString().trim()) {
      return;
    }

    // 에디터 내 선택인지 확인
    const editorBody = document.getElementById('editor-body');
    if (!editorBody.contains(sel.anchorNode)) return;

    selectedRange = sel.getRangeAt(0);
    const rect = selectedRange.getBoundingClientRect();
    const toolbar = document.getElementById('floating-toolbar');

    toolbar.style.top = `${rect.top - 40 + window.scrollY}px`;
    toolbar.style.left = `${rect.left + (rect.width / 2) - 80}px`;
    toolbar.classList.add('visible');
  }

  function hideToolbar() {
    document.getElementById('floating-toolbar').classList.remove('visible');
  }

  function showAiBubble() {
    const toolbar = document.getElementById('floating-toolbar');
    const bubble = document.getElementById('ai-bubble');
    const rect = toolbar.getBoundingClientRect();

    bubble.style.top = `${rect.top - 80 + window.scrollY}px`;
    bubble.style.left = `${rect.left}px`;
    bubble.classList.add('visible');

    const input = document.getElementById('ai-bubble-input');
    input.value = '';
    input.focus();
  }

  function hideAiBubble() {
    document.getElementById('ai-bubble').classList.remove('visible');
  }

  async function handleAiBubbleApply() {
    const instruction = document.getElementById('ai-bubble-input').value.trim();
    if (!instruction || !selectedRange) return;

    const selectedText = selectedRange.toString();
    const apiConfig = Settings.getApiConfig();

    if (!apiConfig.apiKey) {
      App.toast('API 키를 설정하세요.', 'error');
      return;
    }

    hideAiBubble();
    hideToolbar();

    try {
      const result = await AIClient.chat(apiConfig, {
        system: `너는 한시 콘텐츠 작가다. 선택된 텍스트를 지시에 따라 다시 작성해라. 다시 작성한 내용만 출력해라.\n\n${PromptBuilder.BASE_STYLE_RULES}`,
        user: `선택된 텍스트:\n${selectedText}\n\n지시: ${instruction}`,
      });

      // 선택 영역을 결과로 교체
      const sel = window.getSelection();
      sel.removeAllRanges();
      sel.addRange(selectedRange);
      document.execCommand('insertHTML', false, result.replace(/\n/g, '<br>'));

      Editor.syncContentFromDOM();
      App.toast('적용 완료!', 'success');
    } catch (err) {
      App.toast('AI 요청 실패: ' + err.message, 'error');
    }
  }

  // === 슬래시 커맨드 ===
  function initSlashCommand() {
    document.getElementById('editor-body').addEventListener('input', handleSlashInput);
    document.addEventListener('keydown', handleSlashKeydown);

    const palette = document.getElementById('slash-palette');
    palette.innerHTML = `
      <div class="slash-palette__header">빠른 명령</div>
      ${SLASH_COMMANDS.map((cmd, i) =>
        `<div class="slash-palette__item ${i === 0 ? 'selected' : ''}" data-cmd="${cmd.id}">${cmd.icon} ${cmd.label}</div>`
      ).join('')}
      <div class="slash-palette__divider"></div>
      <div class="slash-palette__input-wrap">
        <input class="slash-palette__input" id="slash-custom-input" placeholder="또는 자유롭게 입력...">
      </div>
    `;

    palette.addEventListener('click', e => {
      const item = e.target.closest('.slash-palette__item');
      if (item) executeSlashCommand(item.dataset.cmd);
    });

    document.getElementById('slash-custom-input').addEventListener('keydown', e => {
      if (e.key === 'Enter') {
        executeSlashCommand('custom', e.target.value);
        e.target.value = '';
      }
      if (e.key === 'Escape') hideSlashPalette();
    });
  }

  let slashNode = null; // 슬래시를 입력한 노드

  function handleSlashInput(e) {
    const sel = window.getSelection();
    if (!sel || !sel.anchorNode) return;

    const node = sel.anchorNode;
    const text = node.textContent || '';

    // 빈 줄(또는 줄 시작)에서 / 입력 감지
    if (text.trim() === '/') {
      slashNode = node;
      const range = sel.getRangeAt(0);
      const rect = range.getBoundingClientRect();

      const palette = document.getElementById('slash-palette');
      palette.style.top = `${rect.bottom + 4 + window.scrollY}px`;
      palette.style.left = `${rect.left}px`;
      palette.classList.add('visible');

      // 팔레트 첫 항목 선택
      palette.querySelectorAll('.slash-palette__item').forEach((item, i) => {
        item.classList.toggle('selected', i === 0);
      });
    } else if (!text.includes('/')) {
      hideSlashPalette();
    }
  }

  let selectedCommandIndex = 0;

  function handleSlashKeydown(e) {
    const palette = document.getElementById('slash-palette');
    if (!palette.classList.contains('visible')) return;

    const items = palette.querySelectorAll('.slash-palette__item');

    if (e.key === 'ArrowDown') {
      e.preventDefault();
      selectedCommandIndex = Math.min(selectedCommandIndex + 1, items.length - 1);
      items.forEach((item, i) => item.classList.toggle('selected', i === selectedCommandIndex));
    } else if (e.key === 'ArrowUp') {
      e.preventDefault();
      selectedCommandIndex = Math.max(selectedCommandIndex - 1, 0);
      items.forEach((item, i) => item.classList.toggle('selected', i === selectedCommandIndex));
    } else if (e.key === 'Enter') {
      e.preventDefault();
      const selected = items[selectedCommandIndex];
      if (selected) executeSlashCommand(selected.dataset.cmd);
    } else if (e.key === 'Escape') {
      hideSlashPalette();
    }
  }

  function hideSlashPalette() {
    document.getElementById('slash-palette').classList.remove('visible');
    selectedCommandIndex = 0;
  }

  async function executeSlashCommand(cmdId, customText) {
    hideSlashPalette();

    // 슬래시 텍스트 제거
    if (slashNode) {
      slashNode.textContent = slashNode.textContent.replace('/', '');
    }

    const apiConfig = Settings.getApiConfig();
    if (!apiConfig.apiKey) {
      App.toast('API 키를 설정하세요.', 'error');
      return;
    }

    // 현재 섹션의 내용 가져오기
    const sectionBlock = slashNode?.closest('.section-block');
    const currentContent = sectionBlock?.querySelector('.section-block__content')?.innerHTML || '';

    let instruction;
    switch (cmdId) {
      case 'continue': instruction = '이어서 자연스럽게 작성해줘.'; break;
      case 'rewrite': instruction = '같은 내용을 다른 방식으로 다시 작성해줘.'; break;
      case 'expand': instruction = '더 자세하게, 분량을 늘려서 작성해줘.'; break;
      case 'shorten': instruction = '핵심만 남기고 분량을 줄여줘.'; break;
      case 'tone': instruction = '톤을 좀 더 부드럽고 읽기 편하게 바꿔줘.'; break;
      case 'custom': instruction = customText; break;
      default: return;
    }

    try {
      const result = await AIClient.chat(apiConfig, {
        system: `너는 한시 콘텐츠 작가다. 지시에 따라 작성해라. 작성한 내용만 출력해라.\n\n${PromptBuilder.BASE_STYLE_RULES}`,
        user: `현재 내용:\n${currentContent}\n\n지시: ${instruction}`,
      });

      // 결과를 현재 위치에 삽입
      const contentEl = sectionBlock?.querySelector('.section-block__content');
      if (contentEl) {
        if (cmdId === 'continue') {
          contentEl.innerHTML += '<br>' + result.replace(/\n/g, '<br>');
        } else {
          contentEl.innerHTML = result.replace(/\n/g, '<br>');
        }
      }

      Editor.syncContentFromDOM();
      App.toast('완료!', 'success');
    } catch (err) {
      App.toast('AI 요청 실패: ' + err.message, 'error');
    }
  }

  return { init };
})();
```

```javascript
// tools/ai-writer/js/db.js
'use strict';

const DB = (() => {

  function getConfig() {
    const config = Settings.getSupabaseConfig();
    if (!config.url || !config.key) {
      throw new Error('Supabase URL과 Key를 설정하세요.');
    }
    return config;
  }

  function headers(config) {
    return {
      'apikey': config.key,
      'Authorization': `Bearer ${config.key}`,
      'Content-Type': 'application/json',
      'Prefer': 'return=representation',
    };
  }

  const TABLE_MAP = {
    poem: 'poems',
    poet: 'poets',
    history: 'history_cards',
    article: 'articles',
  };

  const KEY_MAP = {
    poem: 'poem_no_str',
    poet: 'id',
    history: 'id',
    article: 'slug',
  };

  async function save(mode, data) {
    const config = getConfig();
    const table = TABLE_MAP[mode];
    const key = KEY_MAP[mode];
    const keyValue = data[key] || data.id || data.slug;

    if (!table) throw new Error(`알 수 없는 모드: ${mode}`);

    // 기존 레코드 확인
    let existing = null;
    if (keyValue) {
      const checkRes = await fetch(
        `${config.url}/rest/v1/${table}?${key}=eq.${encodeURIComponent(keyValue)}&select=${key}`,
        { headers: headers(config) }
      );
      const checkData = await checkRes.json();
      existing = checkData.length > 0;
    }

    if (existing) {
      // UPDATE
      const res = await fetch(
        `${config.url}/rest/v1/${table}?${key}=eq.${encodeURIComponent(keyValue)}`,
        {
          method: 'PATCH',
          headers: headers(config),
          body: JSON.stringify(data),
        }
      );
      if (!res.ok) throw new Error(`저장 실패: ${res.status} ${await res.text()}`);
      return await res.json();
    } else {
      // INSERT
      const res = await fetch(
        `${config.url}/rest/v1/${table}`,
        {
          method: 'POST',
          headers: headers(config),
          body: JSON.stringify(data),
        }
      );
      if (!res.ok) throw new Error(`저장 실패: ${res.status} ${await res.text()}`);
      return await res.json();
    }
  }

  async function load(mode, id) {
    const config = getConfig();
    const table = TABLE_MAP[mode];
    const key = KEY_MAP[mode];

    const res = await fetch(
      `${config.url}/rest/v1/${table}?${key}=eq.${encodeURIComponent(id)}&select=*`,
      { headers: headers(config) }
    );
    if (!res.ok) throw new Error(`불러오기 실패: ${res.status}`);
    const data = await res.json();
    return data[0] || null;
  }

  // 목록 조회 (불러오기 UI용)
  async function list(mode, limit = 20) {
    const config = getConfig();
    const table = TABLE_MAP[mode];

    const res = await fetch(
      `${config.url}/rest/v1/${table}?select=*&limit=${limit}&order=updated_at.desc`,
      { headers: headers(config) }
    );
    if (!res.ok) throw new Error(`목록 조회 실패: ${res.status}`);
    return await res.json();
  }

  return { save, load, list };
})();
```
